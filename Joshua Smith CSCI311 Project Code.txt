/* Grammer rules
 A sentence is a noun phrase followed by a verb phrase.
 A noun phrase is a determiner followed by a noun.
 A noun phrase can also just me a noun.
 A verb_phrase is a verb followed by a noun phrase.
*/

sentence(sentence(NP, VP)) --> nounPhrase(NP), verbPhrase(VP).
nounPhrase(noun(Det, N)) --> determiner(Det), noun(N).
nounPhrase(noun(N)) --> noun(N).
verbPhrase(verb(V, NP)) --> verb(V), nounPhrase(NP).

determiner(det(the)) --> [the].
determiner(det(a)) --> [a].
noun(noun(josh)) --> [josh].
noun(noun(dog)) --> [dog].
noun(noun(groundhog)) --> [groundhog].
verb(verb(chases)) --> [chases].
verb(verb(sees)) --> [sees].
verb(verb(likes)) --> [likes].
verb(verb(dislikes)) -->[dislikes].

% Base case: Just prints word with current indention.
printTree(Node, Prefix, IsLast) :-
    atomic(Node), !,                                     % If Node is a word.
    ( IsLast -> Branch = '|>> ' ; Branch = '|- ' ),      % Choose branch symbol.
    write(Prefix), write(Branch), writeln(Node).         % Print with prefix.

% Recursive case: Print with increasing indention by 3 spaces.
printTree(Node, Prefix, IsLast) :-
    Node =.. [Label | Children],                         % Breaks Node into Label and Children.
    ( IsLast -> Branch = '|>> ' ; Branch = '|- ' ),      % Choose branch symbol.
    write(Prefix), write(Branch), writeln(Label),        % Print current node.
    
    % Sets new prefix for children.
    ( IsLast -> string_concat(Prefix, '   ', NewPrefix)  % Last child.
              ; string_concat(Prefix, '|  ', NewPrefix)  % Not last.
    ),
    printChildren(Children, NewPrefix).                  % Print children recursively.

% Print a list of children
printChildren([], _).                                    % No children = done.
printChildren([C], Prefix) :-                            % Last child.
    printTree(C, Prefix, true).
printChildren([C|Cs], Prefix) :-                         % Other children.
    printTree(C, Prefix, false),
    printChildren(Cs, Prefix).
